Example

#[macro_use] extern crate polars;
use polars::prelude::*;
use polars::lazy::dsl::*;

fn example() -> Result<()> {
    let df = df! {
            "groups" => &[1, 1, 2, 2, 1, 2, 3, 3, 1],
            "values" => &[1, 2, 3, 4, 5, 6, 7, 8, 8]
        }?;

    let out = df
     .lazy()
     .select(&[
         col("groups"),
         sum("values").over(col("groups")),
     ])
     .collect()?;
    dbg!(&out);
    Ok(())
}
